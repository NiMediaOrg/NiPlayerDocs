# 安装

npm i -S niplayer                                   
pnpm i -S niplayer
yarn add niplayer

# 使用

import Player from "niplayer"

let player = new Player({
    url: "https://novaex.cc/fireworks.mp4", 
    container:document.getElementById("video"),
})
